getColumns Method

Class: Row

Description

This method retrieves a list of columns in a Row object.

Syntax

Row.getColumns()

Parameters

None

 

Returns

A comma-separated list of column definitions. Each column definition takes the form of PSLtype ColumnName

When Became Available

Profile v7.0

Example

type ResultSet rs = Db.select( "NAM,DOB,TYPE", "CIF")

while rs.next() do {

  type Number nCols

 

  // create Row object from ResultSet, get its column list

  type Row rRow = rs.getRow()

  type String sColList = rRow.getColumns()

 

  for nCols = 1:1:sColList.length(",") do {

 

....// format: Type1 Name1,Type2 Name2

    type String sColDef = sColList.piece(",",nCols)

    type String sColTyp = sColDef.piece(" ") // data type

    type String sColNam = sColDef.piece(" ",2) // col. name

    write sColNam, " (", sColTyp,"):",?20,rRow.@sColNam,!

  }

  write !

}
